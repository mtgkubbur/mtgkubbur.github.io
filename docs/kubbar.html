isDark = Generators.observe(notify => {
  const current = () => document.body.classList.contains("quarto-dark");
  notify(current());

  const observer = new MutationObserver(() => {
    notify(current());
  });

  observer.observe(document.body, { attributes: true, attributeFilter: ["class"] });
  return () => observer.disconnect();
})

themeColors = {
  const _ = isDark;
  const bodyStyles = getComputedStyle(document.body);
  const rootStyles = getComputedStyle(document.documentElement);
  const cssVar = (name, fallback = "") =>
    bodyStyles.getPropertyValue(name).trim() ||
    rootStyles.getPropertyValue(name).trim() ||
    fallback;

  return {
    blue: cssVar("--mtg-blue", "#0e68ab"),
    green: cssVar("--mtg-green", "#00733e"),
    red: cssVar("--mtg-red", "#d3202a"),
    gold: cssVar("--mtg-gold", "#8b6914"),
    text: cssVar("--text-primary", "#2a1f14"),
    muted: cssVar("--text-muted", "#5c4f42")
  };
}

viewof selectedCube = Inputs.select(["", ...cubeNames], {
  label: "Kubbur:",
  value: "",
  format: d => d === "" ? "Veldu kubb..." : d
})
{
  if (selectedCube === "") {
    return html`<div class="empty-state">
      <div class="empty-icon">&#9876;</div>
      <div>Veldu kubb til að sjá upplýsingar</div>
    </div>`;
  }

  const summary = cubeSummary.find(d => d.cube === selectedCube);
  const players = perCubePlayer
    .filter(d => d.cube === selectedCube)
    .sort((a, b) => b.winrate - a.winrate || b.total - a.total);
  const cubeTrophies = trophies
    .filter(d => d.cube === selectedCube)
    .sort((a, b) => b.trophies - a.trophies);
  const events = eventResults.filter(d => d.cube === selectedCube);

  const fmtDate = d => {
    const yy = String(d.getFullYear()).slice(2);
    const mm = String(d.getMonth() + 1).padStart(2, '0');
    const dd = String(d.getDate()).padStart(2, '0');
    return `${yy}.${mm}.${dd}`;
  };

  // ── 1. Summary Stats ──
  const summaryCard = html`<div class="player-summary">
    <div class="stat">
      <div class="stat-value" style="color: ${themeColors.blue};">${summary.total_events}</div>
      <div class="stat-label">Viðburðir</div>
    </div>
    <div class="stat">
      <div class="stat-value" style="color: ${themeColors.red};">${summary.total_matches}</div>
      <div class="stat-label">Matches</div>
    </div>
    <div class="stat">
      <div class="stat-value" style="color: ${themeColors.green};">${summary.unique_players}</div>
      <div class="stat-label">Leikmenn</div>
    </div>
    <div class="stat">
      <div class="stat-value" style="color: ${themeColors.gold};">${fmtDate(summary.first_played)}</div>
      <div class="stat-label">Fyrst spilað</div>
    </div>
    <div class="stat">
      <div class="stat-value" style="color: ${themeColors.gold};">${fmtDate(summary.last_played)}</div>
      <div class="stat-label">Síðast spilað</div>
    </div>
  </div>`;

  // ── 2. Trophy Leaderboard ──
  const trophyCard = cubeTrophies.length === 0 ? html`` : html`
    <div class="trophy-card">
      <div class="cube-affinity-title">Bikarhafar</div>
      <table class="trophy-table">
        <thead>
          <tr>
            <th>Leikmaður</th>
            <th>Bikarar</th>
          </tr>
        </thead>
        <tbody>
          ${cubeTrophies.map(t => html`
            <tr>
              <td class="cube-record-name">${t.player}</td>
              <td style="text-align: center; font-weight: 600; color: ${themeColors.gold};">${"🏆".repeat(t.trophies)}</td>
            </tr>
          `)}
        </tbody>
      </table>
    </div>`;

  // ── 3. Player Rankings Table ──
  // Compute match-level stats per player from event results
  const matchByPlayer = new Map();
  events.forEach(d => {
    if (!matchByPlayer.has(d.player)) matchByPlayer.set(d.player, { mw: 0, ml: 0 });
    const s = matchByPlayer.get(d.player);
    s.mw += d.match_wins;
    s.ml += d.matches - d.match_wins;
  });

  const enrichedPlayers = players.map(r => {
    const ms = matchByPlayer.get(r.player) || { mw: 0, ml: 0 };
    const mTotal = ms.mw + ms.ml;
    return { ...r, mw: ms.mw, ml: ms.ml, mTotal, mPct: mTotal > 0 ? Math.round(ms.mw / mTotal * 100) : 0 };
  });

  const rankingsCard = enrichedPlayers.length === 0 ? html`` : html`
    <div class="cube-combined-card">
      <div class="cube-affinity-title">Árangur leikmanna</div>
      <table class="cube-combined-table">
        <thead>
          <tr>
            <th></th>
            <th colspan="3" class="col-group-header">Matches</th>
            <th colspan="3" class="col-group-header">Games</th>
          </tr>
          <tr>
            <th>Leikmaður</th>
            <th>S</th>
            <th>T</th>
            <th>%</th>
            <th>S</th>
            <th>T</th>
            <th>%</th>
          </tr>
        </thead>
        <tbody>
          ${enrichedPlayers.map(r => html`
            <tr>
              <td class="cube-record-name">${r.player}</td>
              <td>${r.mw}</td>
              <td>${r.ml}</td>
              <td style="color: ${r.mPct >= 50 ? themeColors.green : themeColors.red}; font-weight: 600">${r.mPct}%</td>
              <td>${r.wins}</td>
              <td>${r.losses}</td>
              <td style="color: ${r.winrate >= 50 ? themeColors.green : themeColors.red}; font-weight: 600">${r.winrate}%</td>
            </tr>
          `)}
        </tbody>
      </table>
    </div>`;

  return html`<div class="kubbar-content">
    ${summaryCard}
    ${trophyCard}
    ${rankingsCard}
  </div>`;
}
cubeEventDates = {
  if (selectedCube === "") return [];
  const events = eventResults.filter(d => d.cube === selectedCube);
  return [...new Set(events.map(d => +d.date))]
    .sort((a, b) => b - a);
}

viewof selectedEventDate = {
  if (cubeEventDates.length === 0) return Object.assign(html`<span></span>`, { value: "" });

  const fmtDate = ms => {
    const d = new Date(ms);
    const yy = String(d.getFullYear()).slice(2);
    const mm = String(d.getMonth() + 1).padStart(2, '0');
    const dd = String(d.getDate()).padStart(2, '0');
    return `${yy}.${mm}.${dd}`;
  };

  const options = cubeEventDates.map(ms => ({ value: String(ms), label: fmtDate(ms) }));

  return Inputs.select(options.map(o => o.value), {
    label: "Viðburður:",
    value: options[0].value,
    format: v => options.find(o => o.value === v)?.label || v
  });
}
{
  if (selectedCube === "" || !selectedEventDate || cubeEventDates.length === 0) return html``;

  const dateMs = +selectedEventDate;
  const eventPlayers = eventResults
    .filter(d => d.cube === selectedCube && +d.date === dateMs)
    .sort((a, b) => b.match_wins - a.match_wins);

  if (eventPlayers.length === 0) return html``;

  return html`<div class="kubbar-content">
    <div class="recent-events-card">
      <div class="cube-affinity-title">Viðburður</div>
      <table class="trophy-table">
        <thead>
          <tr>
            <th></th>
            <th colspan="2" class="col-group-header">Matches</th>
            <th colspan="2" class="col-group-header">Games</th>
          </tr>
          <tr>
            <th>Leikmaður</th>
            <th>S</th>
            <th>T</th>
            <th>S</th>
            <th>T</th>
          </tr>
        </thead>
        <tbody>
          ${eventPlayers.map(p => html`
            <tr>
              <td class="cube-record-name">${p.player}${p.match_wins >= 3 ? " 🏆" : ""}</td>
              <td style="text-align: center">${p.match_wins}</td>

              <td style="text-align: center">${p.game_wins}</td>
              <td style="text-align: center">${p.game_losses}</td>
            </tr>
          `)}
        </tbody>
      </table>
    </div>
  </div>`;
}
