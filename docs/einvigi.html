isDark = {
  let dark = document.body.classList.contains("quarto-dark");
  yield dark;
  while (true) {
    await new Promise(r => setTimeout(r, 150));
    const now = document.body.classList.contains("quarto-dark");
    if (now !== dark) { dark = now; yield dark; }
  }
}
