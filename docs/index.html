ELO-stigaröðun

Uppfært 2026-02-19

viewof cubeFilter = Inputs.select(
  ["Allt", "High", "Medium", "Low", "Annað"],
  {
    label: "Kubbaflokkur:",
    value: "Allt"
  }
)
{
  const panels = document.querySelectorAll(".cube-table-panel");
  const key = cubeFilter === "Annað" ? "Other" : cubeFilter === "Allt" ? "Allt" : cubeFilter;
  panels.forEach(el => {
    el.classList.toggle("active", el.dataset.cube === key);
  });
}
