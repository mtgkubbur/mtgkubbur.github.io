isDark = Generators.observe(notify => {
  const current = () => document.body.classList.contains("quarto-dark");
  notify(current());

  const observer = new MutationObserver(() => {
    notify(current());
  });

  observer.observe(document.body, { attributes: true, attributeFilter: ["class"] });
  return () => observer.disconnect();
})

themeColors = {
  const _ = isDark;
  const bodyStyles = getComputedStyle(document.body);
  const rootStyles = getComputedStyle(document.documentElement);
  const cssVar = (name, fallback = "") =>
    bodyStyles.getPropertyValue(name).trim() ||
    rootStyles.getPropertyValue(name).trim() ||
    fallback;

  return {
    blue: cssVar("--mtg-blue", "#0e68ab"),
    green: cssVar("--mtg-green", "#00733e"),
    red: cssVar("--mtg-red", "#d3202a"),
    gold: cssVar("--mtg-gold", "#8b6914"),
    text: cssVar("--text-primary", "#2a1f14"),
    muted: cssVar("--text-muted", "#5c4f42")
  };
}

{
  const fmtDate = d => {
    const dd = String(d.getDate()).padStart(2, '0');
    const mm = String(d.getMonth() + 1).padStart(2, '0');
    const yyyy = d.getFullYear();
    return `${dd}.${mm}.${yyyy}`;
  };

  const clean = v => (v === undefined || v === null || v === "undefined" || v === "NA") ? "" : v;

  const nextDate = upcoming.length > 0 ? +upcoming[0].date : null;
  const isNext = d => nextDate !== null && +d === nextDate;

  // ── Upcoming events table ──
  let upcomingSection;
  if (upcoming.length === 0) {
    upcomingSection = html`<div class="empty-state">
      <div class="empty-icon">&#128197;</div>
      <div>Engin viðburðir á dagskrá</div>
    </div>`;
  } else {
    upcomingSection = html`<div class="calendar-table-card">
      <div class="cube-affinity-title">Væntanlegir viðburðir</div>
      <table class="trophy-table">
        <thead>
          <tr>
            <th>Dagsetning</th>
            <th>Kubbur</th>
            <th>Eigandi</th>
          </tr>
        </thead>
        <tbody>
          ${upcoming.map(e => html`
            <tr class="${isNext(e.date) ? "calendar-next-event" : ""}">
              <td>${fmtDate(e.date)}</td>
              <td>${clean(e.cube)}</td>
              <td>${clean(e.host)}</td>
            </tr>
          `)}
        </tbody>
      </table>
    </div>`;
  }

  // ── Past events (collapsible) ──
  let pastSection = html``;
  if (past.length > 0) {
    pastSection = html`<div class="calendar-table-card" style="margin-top: 1.5rem;">
      <details class="calendar-past-toggle">
        <summary>Liðnir viðburðir (${past.length})</summary>
        <table class="trophy-table">
          <thead>
            <tr>
              <th>Dagsetning</th>
              <th>Kubbur</th>
              <th>Eigandi</th>
              <th>Leikmenn</th>
            </tr>
          </thead>
          <tbody>
            ${past.map(e => html`
              <tr>
                <td>${fmtDate(e.date)}</td>
                <td>${clean(e.cube)}</td>
                <td>${clean(e.host)}</td>
                <td style="text-align: center;">${clean(e.players) || "—"}</td>
              </tr>
            `)}
          </tbody>
        </table>
      </details>
    </div>`;
  }

  return html`<div class="calendar-content">
    ${upcomingSection}
    ${pastSection}
  </div>`;
}
